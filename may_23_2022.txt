//find missing element in array

#include<stdio.h>

int main()
{
int a[5]={1,2,5,3};
int sum=0,i;
for(i=0;i<5;i++)
{
sum=sum+a[i];
}
int ele=(5*(5+1))/2-(sum);
printf("missing element=%d",ele);

}
